3. 单词接龙：深度遍历模板
````
【广度优先模板】
if indexOf(endWord, wordList) == -1 {
    return 0
}
step := 0
visited := make([]bool, len(wordList))
queue := []string{beginWord}
for len(queue) > 0 {
    l := len(queue)
    step++
    for i := 0; i < l; i++ {
        if queue[i] == endWord {
            return step
        }
        for j := 0; j < len(wordList); j++ {
            if !visited[j] && oneDiff(queue[i], wordList[j]) {
                queue = append(queue, wordList[j])
                visited[j] = true
            }
        }
    }
    queue = queue[l:]
}
return 0
````
